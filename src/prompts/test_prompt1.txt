// Import the necessary modules and functions
import {{ assertEquals }} from "https://deno.land/std/testing/asserts.ts";

// Import the function to be tested
import {{ capitalize }} from "../code/temp.js";

// Define the test cases
Deno.test("capitalize should capitalize the first letter of a word", () => {
  // Input
  const text = "hello";

  // Expected output
  const expected = "Hello";

  // Actual output
  const actual = capitalize(text);

  // Assert the expected and actual values are equal
  assertEquals(actual, expected);
});

Deno.test("capitalize should return an empty string when input is empty", () => {
  // Input
  const text = "";

  // Expected output
  const expected = "";

  // Actual output
  const actual = capitalize(text);

  // Assert the expected and actual values are equal
  assertEquals(actual, expected);
});
